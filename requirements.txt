# =============================================================================
# Fashion Trend Assistant - Python Dependencies
#
# To install all dependencies, run the following command in your
# activated virtual environment:
# pip install -r requirements.txt
# =============================================================================

# -----------------------------------------------------------------------------
# Core Application & Data
# -----------------------------------------------------------------------------

# For loading environment variables from the .env file into the application.
python-dotenv

# For data validation and settings management. Ensures the data from the LLM
# matches our predefined Pydantic models.
pydantic

# -----------------------------------------------------------------------------
# Google API Clients
# -----------------------------------------------------------------------------

# The official Python SDK for the Google Gemini API.
# This is the primary library for all LLM interactions (text generation,
# summarization, and embedding generation).
google-genai

# The Google API Client Library for Python. This is used specifically for
# the Custom Search JSON API to find authoritative sources in the research phase.
google-api-python-client

# -----------------------------------------------------------------------------
# Web Scraping & Automation
# -----------------------------------------------------------------------------

# An advanced browser automation tool that allows us to render JavaScript-heavy
# websites, which is crucial for modern web scraping.
playwright

# A powerful and popular library for parsing HTML and XML documents. We use it
# to intelligently extract the main article content from the HTML fetched by Playwright.
beautifulsoup4

# -----------------------------------------------------------------------------
# Vector Database Caching
# -----------------------------------------------------------------------------

# The vector database used to create a local, persistent, semantic cache of
# trend reports, enabling instant retrieval of similar past results.
chromadb

# -----------------------------------------------------------------------------
# Networking Utilities
# -----------------------------------------------------------------------------

# A simple and robust library for making standard HTTP requests. Used here for
# the IP geolocation API call in the location_helper.
requests

# A stealth plugin for Playwright to help evade bot detection mechanisms.
playwright-stealth
